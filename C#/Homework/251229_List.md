# 과제 1
```csharp
using System;
using System.Collections.Generic;

class Program
{
    public static void Main(string[] arg)
    {
        VendingMachine vm = new VendingMachine();
        while(true)
        {
            Console.WriteLine("행동을 선택하세요");
            Console.WriteLine("1. 우유 넣기 2. 우유 빼기 3. 우유 정보 출력" );

            int input = int.Parse(Console.ReadLine());

            if(input == 1)
            {
                Console.WriteLine("생산일자 6자리를 입력하세요!");
                int ymd = int.Parse(Console.ReadLine());
                Milk milk = new Milk(ymd);
                vm.PutInMilk(milk);
            }

            if(input == 2)
            {
                vm.PutOutMilk();
            }

            if(input == 3)
            {
                vm.PrintInfo();
            }
        }
    }
}

 class Milk
{
    public Milk(int ymd = 000000)
    {
        this.Ymd = ymd;
    }

     public int Ymd { get; private set; }
}
    
class VendingMachine
{
    public VendingMachine()
    {
    }

    public void PutInMilk(Milk milk)
    {
        Milks.Enqueue(milk);
    }
    public void PutOutMilk()
    {
        if(Milks.Count == 0)
        {
            Console.WriteLine($"남은 우유가 없습니다!");
        }
         Milks.Dequeue();
        Console.WriteLine($"남은 우유는 {Milks.Count}개 입니다.");
    }

    public void PrintInfo()
    {
        Console.WriteLine("\n--- 현재 자판기 우유 목록 ---");
        if (Milks.Count == 0)
        {
            Console.WriteLine("자판기가 비어 있습니다.");
            return;
        }

        int count = 1;
        foreach (Milk milk in Milks)
        {
            // milk.Ymd 프로퍼티를 통해 생산일자에 접근
            Console.WriteLine($"{count}번 우유 [생산일자: {milk.Ymd}]");
            count++;
        }
        Console.WriteLine("---------------------------\n");
    }

    public Queue<Milk> Milks = new Queue<Milk>();
}
```

# 과제 2
```csharp
using System;
using System.Collections.Generic;

enum Shape
{
    Spade,
    Heart,
    Clover,
    Diamond
}
class Program
{
    public static void Main(string[] arg)
    {
        CardDeck newDeck = new CardDeck();

        Console.WriteLine("카드풀이 만들어졌습니다.\n 어떤 행동을 하시겠습니까?");

        int input = 0;

        while(true)
        {
            Console.WriteLine("1. 맨위의 카드보기 2. 한장 뽑기");

            input = int.Parse(Console.ReadLine());

            if(input == 1)
            {
                newDeck.ShowTopCard();
            }

            if(input == 2)
            {
                Card pick = newDeck.DrawCard();

                if(pick != null)
                {
                    Console.WriteLine($"뽑은 카드: {pick.GetInfo()} (usedCards에 들어갔습니다)");
                }
            }
        }
    }
}

class Card
{
    public Shape shape { get; }
    public int num { get; }

    public Card(Shape shape, int num)
    {
        this.shape = shape;
        this.num = num;
    }
    public string GetInfo()
    {
        string displayNum = num switch { 1 => "A", 11 => "J", 12 => "Q", 13 => "K", _ => num.ToString() };
        return $"{shape}({displayNum})";
    }
}

class CardDeck
{
    private Stack<Card> unusedCards = new Stack<Card>();
    private Stack<Card>   usedCards = new Stack<Card>();

    public CardDeck()
    {
        Random rand = new Random();
        List<Card> tempPool = new List<Card>();

        for (int s = 0; s < 4; s++)
        {
            for (int v = 1; v <= 13; v++)
            {
                tempPool.Add(new Card((Shape)s, v));
            }
        }

        while (tempPool.Count > 0)
        {
            int index = rand.Next(0, tempPool.Count);
            unusedCards.Push(tempPool[index]);
            tempPool.RemoveAt(index);
        }
    }

    public void ShowTopCard()
    {
        if (unusedCards.Count > 0)
            Console.WriteLine($"맨위의 카드 확인 : {unusedCards.Peek().GetInfo()}");
        else
            Console.WriteLine("남은 카드가 없다.");
    }

    public Card DrawCard()
    {
        if(unusedCards.Count > 0)
        {
            Card card = unusedCards.Pop();
            usedCards.Push(card);
            return card;
        }

        return null;
    }
}
```
