# 과제 2. 아이템과 인벤토리 구현
```csharp
using System;
using System.Collections.Generic;
using System.Collections.Specialized;
using System.Runtime.InteropServices;
using System.Security.Cryptography.X509Certificates;

class Program
{
    enum Items
    {
        Weapon = 1,
        Armor,
        HeadPiece,
        Shoes,
        Gloves
    }
    struct Item
    {
        Items type;

        public Item(Items type)
        {
            this.type = type;
        }

        public Items GetItemType()
        {
            return type;
        }

        public void PrintName()
        {
            Console.WriteLine
            (type);
        }
    }

    class Inventory
    {
        Dictionary<Items, int> items;

        public Inventory()
        {
            items = new Dictionary<Items, int>();
        }

        public void AddItem(Item item)
        {
            Items type = item.GetItemType();

            if (items.ContainsKey(type))
            {
                items[type]++;
            }
            else
            {
                items[type] = 1;
            }
        }

        public void DiscardItem(Item item)
        {
            Items type = item.GetItemType();

            if (!items.ContainsKey(type))
                return;

            items[type]--;

            // 개수가 0이면 제거
            if (items[type] <= 0)
            {
                items.Remove(type);
            }
        }

        public void PrintAllItem()
        {
            foreach (var pair in items)
            {
                Console.WriteLine($"{pair.Key} : {pair.Value}개");
            }
        }
    }


    static void Main(string[] args)
    {
        for (int i = 1; i <= (int)Items.Gloves; i++)
        {
            Console.WriteLine($"{i}. {(Items)i}");
        }

        Console.WriteLine("총 5개의 아이템을 선택합니다.");

        Inventory inventory = new Inventory();

        for (int i = 0; i < 5; i++)
        {
            Console.Write("아이템 번호 입력: ");
            int input = int.Parse(Console.ReadLine());

            Item item = new Item((Items)input);
            inventory.AddItem(item);
        }

        Console.WriteLine("=== 인벤토리 ===");
        inventory.PrintAllItem();

        { 
            Console.WriteLine("하나를 추가합니다.");
            int input = int.Parse(Console.ReadLine());
            Item NewItem = new Item((Items)input);
            inventory.AddItem(NewItem);
        }

        {
            Console.WriteLine("하나를 버립니다.");
            int input = int.Parse(Console.ReadLine());
            Item DiscardItem = new Item((Items)input);
            inventory.DiscardItem(DiscardItem);
        }

        Console.WriteLine("=== 인벤토리 ===");
        inventory.PrintAllItem();
    }
}
```

### 바이트 패딩이란?
컴파일러는 성능향상을 위해 CPU가 접근하기 쉬운 위치에 필드를 배치한다. 그러다보니 중간에 빈 공간이 들어가게 되는데 이를 바이트 패딩이라고 한다.
