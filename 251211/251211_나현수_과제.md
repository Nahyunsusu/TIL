```c#
{

    {

        {

            {
                Console.Write($"{i + 1}번 요소를 입력해 주십시오 : ");
                string temp = Console.ReadLine();
                int.TryParse(temp, out arr[i]);
            }

            for(int i=0;i<arr.Length;i++)
            {
                Console.WriteLine($"{i + 1}번째 요소 {arr[i]}");
            }
        }
    }
```

# 과제 2. 가장 큰 수 구하기
```c#
using System;
using System.Collections.Generic;
using System.Collections.Specialized;
using System.Runtime.InteropServices;
using System.Security.Cryptography.X509Certificates;

namespace _1209_Temp
{
    public static class SwitchExample
    {
        public static void Main()
        {
            int[] arr = new int[4];

            for(int i=0;i<arr.Length;i++)
            {
                Console.Write($"{i + 1}번 요소를 입력해 주십시오 : ");
                string temp = Console.ReadLine();
                int.TryParse(temp, out arr[i]);
            }

            Console.WriteLine("입력하신 요소는 다음과 같습니다.");
            for(int i=0;i<arr.Length;i++)
            {
                Console.WriteLine($"{i + 1}번째 요소 {arr[i]}");
            }

            int max = arr[0];
            for (int i = 1; i < arr.Length; i++)
            {
                if (arr[i] > max)
                {
                    max = arr[i];
                }
            }

            Console.Write($"가장 큰수는 {max}입니다.");
        }
    }
}
```

# 과제 3. 다차원 배열과 함수 다루기
```c#
using System;
using System.Collections.Generic;
using System.Collections.Specialized;
using System.Runtime.InteropServices;
using System.Security.Cryptography.X509Certificates;

namespace _1209_Temp
{
    public static class SwitchExample
    {
        public static void Main()
        {
            int[,] arr = new int[4,4];

            int count = 1;
            
            for(int i=0;i<arr.GetLength(0);i++)
            {
                for(int j=0;j<arr.GetLength(1);j++)
                {
                    arr[i, j] = 3 * count;
                    count++;
                }
            }

            for (int i = 0; i < arr.GetLength(0); i++)
            {
                for (int j = 0; j < arr.GetLength(1); j++)
                {
                    Console.Write(arr[i, j]);
                    Console.Write("\t");
                }
                Console.WriteLine(" ");
            }
        }
    }
}
```

# 과제 5. Main함수 조사
<span style="color:Aqua ">메인 함수의 구조</span>
```c#
using System;

class Program
{
    static void Main(string[] args)
    {
        // 프로그램 실행 시 수행될 코드
    }
}
```
- using System; : .NET Framework의 기본 클래스 라이브러리를 사용하기 위한 선언.
- class Program : Program이라는 클래스를 선언한다. 프로그램의 이름은 다르게 지정할 수 있다.
- static void Main(string[] args) : 프로그램의 시작점인 Main 함수를 선언한다.

그렇다면 Main함수의 매개변수인 <span style="color:Aqua ">args</span>는 무엇인가?

-> 프로그램을 실행할 대 전달되는 명령 줄 인수를 담고 있다.
